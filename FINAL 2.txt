    function makeThumb(src, label){
      const el = document.createElement('div');
      el.className = 'thumb';
      const img = document.createElement('img');
      img.src = src;
      img.alt = label||'thumb';
      el.appendChild(img);
      return el;

      const track = document.getElementById(trackId);

    // WhatsApp number (replace if needed)
    const WA = '923200436201';

    function escapeVal(v){ return encodeURIComponent(v||''); }

    function handleSend(e){
      e.preventDefault();
      const name = document.getElementById('name').value.trim();
      const age = document.getElementById('age').value.trim();
      const country = document.getElementById('country').value.trim();
      const phone = document.getElementById('phone').value.trim();
      const reason = document.getElementById('reason').value.trim();
      const channel = document.getElementById('channel').value.trim();
      const insta = document.getElementById('insta').value.trim();
      const email = document.getElementById('email').value.trim();
      const budget = document.getElementById('budget').value;
      const type = document.getElementById('type').value;
      const note = document.getElementById('note').value.trim();

      // required-check: name, age, country, budget
      if(!name || !age || !country || !budget){
        setStatus('Please fill Name, Age, Country and Budget to send.', true);
        return false;
      }

      // build message (human readable)
      let message = 'New Inquiry from Jefix.Design%0A';
      message += 'Name: ' + escapeVal(name) + '%0A';
      message += 'Age: ' + escapeVal(age) + '%0A';
      message += 'Country: ' + escapeVal(country) + '%0A';
      if(phone) message += 'Phone: ' + escapeVal(phone) + '%0A';
      if(reason) message += 'Reason: ' + escapeVal(reason) + '%0A';
      if(channel) message += 'Channel: ' + escapeVal(channel) + '%0A';
      if(insta) message += 'Instagram: ' + escapeVal(insta) + '%0A';
      if(email) message += 'Email: ' + escapeVal(email) + '%0A';
      message += 'Budget: ' + escapeVal(budget) + '%0A';
      message += 'Type: ' + escapeVal(type) + '%0A';
      if(note) message += 'Note: ' + escapeVal(note) + '%0A';

      // open whatsapp in new tab (private message to your number)
      const url = 'https://wa.me/' + WA + '?text=' + message;
      window.open(url, '_blank');

      setStatus('Opening WhatsApp — message ready (private).', false);
      document.getElementById('inqForm').reset();
      return false;
    }

    function setStatus(t, err){
      const s = document.getElementById('status');
      s.textContent = t || '';
      s.style.color = err ? '#ffb4b4' : '#b7f7ef';
      setTimeout(()=> s.textContent = '', 6000);
    }
